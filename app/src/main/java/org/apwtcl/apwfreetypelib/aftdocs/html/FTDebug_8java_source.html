122  public static void Debug(int level, DebugTag type, String tag, String str) throws IndexOutOfBoundsException {
123  if (type.getVal() < 0 || type.getVal() >= DBG_MAX_TYPES) {
124  throw new IndexOutOfBoundsException(String.format("bad type: %d in FTShowDebug", type));
125  }
126  if (debug_types[type.getVal()] && level >= debug_levels[type.getVal()]) {
