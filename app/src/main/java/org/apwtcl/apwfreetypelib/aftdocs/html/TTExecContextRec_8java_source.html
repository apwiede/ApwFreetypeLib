190  /* =====================================================================
191  * TTLoadContext
192  *
193  * <Description>
194  * Prepare an execution context for glyph hinting.
195  *
196  * <Input>
197  * face :: A handle to the source face object.
198  *
199  * size :: A handle to the source size object.
200  *
201  * <Return>

204  * <Note>
205  * Only the glyph loader and debugger should call this function.
206  *
207  * =====================================================================
208  */
209  public FTError.ErrorTag TTLoadContext(TTFaceRec face, TTSizeRec size) {
210  Debug(0, DebugTag.DBG_LOAD_GLYPH, TAG, "TTLoadContext");
211  FTError.ErrorTag error = FTError.ErrorTag.ERR_OK;
212  int i;
213  TTMaxProfile maxp ;
214 
215  this.face = face;
216  maxp = face.max_profile;
217  this.size = size;
218  if (size != null) {
219 Debug(0, DebugTag.DBG_LOAD_GLYPH, TAG, "num_function_defs: "+size.num_function_defs+"!"+numFDefs+"!");
220 Debug(0, DebugTag.DBG_LOAD_GLYPH, TAG, "max_function_defs: "+size.max_function_defs+"!"+maxFDefs+"!");
221 Debug(0, DebugTag.DBG_LOAD_GLYPH, TAG, "function_defs: "+size.function_defs+"=="+FDefs+"!");
222  numFDefs = size.num_function_defs;
223  maxFDefs = size.max_function_defs;
224  numIDefs = size.num_instruction_defs;
225  maxIDefs = size.max_instruction_defs;
226  FDefs = size.function_defs;
227  IDefs = size.instruction_defs;
228  tt_metrics = size.ttmetrics;
229  metrics = size.metrics;
230  maxFunc = size.max_func;
231  maxIns = size.max_ins;
232  for (i = 0; i < TTSizeRec.TT_MAX_CODE_RANGES; i++) {
233  codeRangeTable[i] = size.codeRangeTable[i];
234  }
235  /* set graphics state */
236  graphics_state = size.GS;
237  cvtSize = (int)size.cvt_size;
238  cvt = size.cvt;
239  storeSize = size.storage_size;
240  storage = size.storage;
241  twilight = size.twilight;
242  /* In case of multi-threading it can happen that the old size object */
243  /* no longer exists, thus we must clear all glyph zone references. */
244  // ft_memset( &exec.zp0, 0, sizeof ( exec.zp0 ) );
245  // exec.zp1 = exec.zp0;
246  // exec.zp2 = exec.zp0;
247  // ATTENTION!! have to call reset otherwise the references are wrong in
248  // for example DirectMoveX!!
249  zp0.reset();
250  zp1.reset();
251  zp2.reset();
252  }
253  /* XXX: We reserve a little more elements on the stack to deal safely */
254  /* with broken fonts like arialbs, courbs, timesbs, etc. */
255 Debug(0, DebugTag.DBG_LOAD_GLYPH, TAG, "face.id: "+face+"!"+face.max_profile+"!");
256  stackSize = face.max_profile.maxStackElements + 32;
257  // tmp = exec.stackSize;
258 // error = UpdateMax(exec.memory, &tmp, sizeof ( FT_F26Dot6 ), (void*)&exec.stack, maxp.maxStackElements + 32 );
259  int[] tmp1 = null;
260  if (stack != null) {
261  tmp1 = new int[stackSize];
262  tmp1 = java.util.Arrays.copyOf(stack, stackSize);
263  }
264  stack = new int[maxp.maxStackElements + 32];
265  if (tmp1 != null) {
266  stack = java.util.Arrays.copyOf(tmp1, stackSize);
267  }
268  for (int k = stackSize; k < maxp.maxStackElements + 32; k++) {
269  stack[k] = 0;
270  }
271  stackSize = maxp.maxStackElements + 32;
272  if (stackSize > stack.length) {
273  Log.e(TAG, "exec.stackSize > exec.stack!!");
274  error = FTError.ErrorTag.UNEXPECTED_NULL_VALUE;
275  return error;
276  }
277  if (error != FTError.ErrorTag.ERR_OK) {
278  return error;
279  }
280  glyphSize = face.max_profile.maxSizeOfInstructions;
281 // tmp = exec.glyphSize;
282 // error = Update_Max(exec.memory, &tmp, sizeof ( FT_Byte ), (void*)&exec.glyphIns, maxp.maxSizeOfInstructions );
283  TTOpCode.OpCode[] tmp2 = null;
284  if (glyphIns != null) {
285  tmp2 = new TTOpCode.OpCode[glyphSize];
286  tmp2 = java.util.Arrays.copyOf(glyphIns, glyphSize);
287  }
288  glyphIns = new TTOpCode.OpCode[maxp.maxSizeOfInstructions];
289  if (tmp2 != null) {
290  glyphIns = java.util.Arrays.copyOf(tmp2, glyphSize);
291  }
292  for (int j = glyphSize; j < maxp.maxSizeOfInstructions; j++) {
293  glyphIns[j] = TTOpCode.OpCode.SVTCA_y;
294  }
295  glyphSize = maxp.maxSizeOfInstructions;
296  if (glyphSize > glyphIns.length) {
297  Log.e(TAG, "exec.glyphSize > exec.glyphIns!!");
298  error = FTError.ErrorTag.UNEXPECTED_NULL_VALUE;
299  return error;
300  }
301  if (error != FTError.ErrorTag.ERR_OK) {
302  return error;
303  }
304  // exec.zp1 = exec.pts;
305  // exec.zp2 = exec.pts;
306  // exec.zp0 = exec.pts;
307  zp1.copy(pts);
308  zp2.copy(pts);
309  zp0.copy(pts);
310  pts.n_points = 0;
311  pts.n_contours = 0;
312 
313  instruction_trap = false;
314  return error;
315  }
