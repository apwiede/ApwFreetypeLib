129  private FTError.ErrorTag FTAddModule(FTLibraryRec library, FTModuleClassRec module_clazz) {
130  FTError.ErrorTag error = FTError.ErrorTag.ERR_OK;
131  int nn;
132  FTModuleRec module = null;
133 
134  if (library == null) {
135  return FTError.ErrorTag.INIT_INVALID_LIBRARY_HANDLE;
136  }
137  if (module_clazz == null) {
138  return FTError.ErrorTag.INIT_INVALID_ARGUMENT;
139  }
140 
141  for (nn = 0; nn < library.num_modules; nn++) {
142  module = library.modules[nn];
143 FTDebug.Debug(0, FTDebug.DebugTag.DBG_INIT, TAG, "nn: "+nn+"!"+module.module_clazz);
144  if (module.module_clazz.module_name.equals(module_clazz.module_name)) {
145  /* this installed module has the same name, compare their versions */
146  if (module_clazz.module_version <= module.module_clazz.module_version) {
147  return FTError.ErrorTag.INIT_LOWER_MODULE_VERSION;
148  }
149  /* remove the module from our list, then exit the loop to replace */
150  /* it by our new version.. */
151 // FTRemoveModule(library, module);
152  break;
153  }
154  }
155  error = FTError.ErrorTag.ERR_OK;
156  if (library.num_modules >= FT_MAX_MODULES) {
157  error = FTError.ErrorTag.INIT_TOO_MANY_DRIVERS;
158  return error;
159  }
160  FTRendererRec renderer = null;
161  TTDriverRec driver = null;
162  /* allocate module object */
163  switch (module_clazz.module_type) {
164  case UNKNOWN:
165  Log.e(TAG, "module type unknown in FTAddModule: "+module_clazz.module_type.getDescription());
166  return FTError.ErrorTag.INIT_BAD_ARGUMENT;
167  case FT_MODULE:
168  module = new FTModuleRec();
169  break;
170  case FT_RENDERER:
171  module = new FTRendererRec();
172  renderer = (FTRendererRec)module;
173  renderer.clazz = (FTRendererClassRec)module_clazz;
174  break;
175  case TT_DRIVER:
176  module = new TTDriverRec();
177  driver = (TTDriverRec)module;
178  break;
179  default:
180  Log.e(TAG, String.format("unknown module type in FTAddModule: %d", module_clazz.module_type.getVal()));
181  return FTError.ErrorTag.INIT_BAD_ARGUMENT;
182  }
183  // base initialization
184  module.library = library;
185  module.module_clazz = module_clazz;
186  /* check whether the module is a renderer - this must be performed */
187  /* before the normal module initialization */
188  int flags = module.module_clazz.module_flags;
189  if (Flags.Module.isRenderer(flags)) {
190 FTDebug.Debug(0, FTDebug.DebugTag.DBG_INIT, TAG, "FTAddModule module is renderer: "+module.module_clazz.module_name+" "+Flags.Module.RENDERER);
191  // add to the renderers list
192 FTDebug.Debug(0, FTDebug.DebugTag.DBG_INIT, TAG, "REND1: "+module.module_clazz.module_name);
193  error = renderer.addRenderer();
194  if (error != FTError.ErrorTag.ERR_OK) {
195  if (Flags.Module.isFontDriver(flags)) {
196  if (Flags.Module.isDriverNoOutlines(flags)) {
197  FTGlyphLoaderRec.FTGlyphLoaderDone(driver.glyph_loader);
198  }
199  }
200  if (Flags.Module.isRenderer(flags)) {
201  if (renderer.clazz != null && renderer.glyph_format == FTTags.GlyphFormat.OUTLINE &&
202  renderer.raster != null) {
203  renderer.clazz.rasterDone(renderer.raster);
204  }
205  }
206 // FT_FREE(module);
207  return error;
208  }
209  }
210  // is the module a auto-hinter?
211  if (Flags.Module.isHinter(flags)) {
212  library.auto_hinter = module;
213  }
214  // if the module is a font driver
215  if (Flags.Module.isFontDriver(flags)) {
216 FTDebug.Debug(0, FTDebug.DebugTag.DBG_INIT, TAG, "FT_Add_Module module is font driver: "+module.module_clazz.module_name+" "+Flags.Module.FONT_DRIVER);
217  // allocate glyph loader if needed
218  driver.driver_clazz = (FTDriverClassRec)module.module_clazz;
219  int driver_flags = driver.module_clazz.module_flags;
220  if (! Flags.Module.isDriverNoOutlines(driver_flags)) {
221 FTDebug.Debug(0, FTDebug.DebugTag.DBG_INIT, TAG, "FT_Add_Module module uses outlines: "+module.module_clazz.module_name+" "+Flags.Module.FONT_DRIVER);
222  driver.glyph_loader = new FTGlyphLoaderRec();
223  }
224  }
225 FTDebug.Debug(0, FTDebug.DebugTag.DBG_INIT, TAG, "FT_Add_Module call module_init for module: "+module.module_clazz.module_name);
226  error = module_clazz.moduleInit(module);
227  if (error == FTError.ErrorTag.ERR_OK) {
228  /* add module to the library's table */
229  library.modules[library.num_modules++] = module;
230  return error;
231  }
232  if (Flags.Module.isFontDriver(flags)) {
233  driver.module_clazz = module.module_clazz;
234  if (!Flags.Module.isDriverNoOutlines(flags)) {
235  FTGlyphLoaderRec.FTGlyphLoaderDone(driver.glyph_loader);
236  }
237  }
238  if (Flags.Module.isRenderer(flags)) {
239  renderer.module_clazz = module.module_clazz;
240  if (renderer.clazz != null && renderer.glyph_format == FTTags.GlyphFormat.OUTLINE &&
241  renderer.raster != null) {
242  renderer.clazz.rasterDone(renderer.raster);
243  }
244  }
245  return error;
246  }
