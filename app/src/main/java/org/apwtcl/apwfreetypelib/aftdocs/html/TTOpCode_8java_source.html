316  public static OpCode getTableTag(int i) {
317  if (tagToOpCodeMapping == null) {
318  initMapping();
319  }
320  return tagToOpCodeMapping.get(i);
321  }
322  private static void initMapping() {
323  tagToOpCodeMapping = new SparseArray<OpCode>();
324  for (OpCode t : values()) {
325  tagToOpCodeMapping.put(t.val, t);
326  }
327  }
