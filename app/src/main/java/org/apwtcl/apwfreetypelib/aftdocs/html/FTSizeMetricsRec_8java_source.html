69  public static FTError.ErrorTag FTRequestMetrics(FTFaceRec face, FTSizeRequestRec req) {
70  FTError.ErrorTag error = FTError.ErrorTag.ERR_OK;
71  boolean doall = true;
72 
73 Debug(0, DebugTag.DBG_INIT, TAG, "FTRequestMetrics:");
74  FTSizeMetricsRec metrics;
75 
76  metrics = face.size.metrics;
77  if ((face.face_flags & Flags.Face.SCALABLE.getVal()) != 0) {
78  int w = 0;
79  int h = 0;
80  int scaled_w = 0;
81  int scaled_h = 0;
82 
83  switch (req.type) {
84  case NOMINAL:
85  h = face.units_per_EM;
86  w = h;
87 Debug(0, DebugTag.DBG_INIT, TAG, "met1: "+w+"!"+h+"!");
88  break;
89  case REAL_DIM:
90  h = face.ascender - face.descender;
91  w = h;
92 Debug(0, DebugTag.DBG_INIT, TAG, "met2: "+w+"!"+h+"!");
93  break;
94  case BBOX:
95  w = face.bbox.xMax - face.bbox.xMin;
96  h = face.bbox.yMax - face.bbox.yMin;
97 Debug(0, DebugTag.DBG_INIT, TAG, "met3: "+w+"!"+h+"!");
98  break;
99  case CELL:
100  w = face.max_advance_width;
101  h = face.ascender - face.descender;
102 Debug(0, DebugTag.DBG_INIT, TAG, "met4: "+w+"!"+h+"!");
103  break;
104  case SCALES:
105  metrics.x_scale = req.width;
106  metrics.y_scale = req.height;
107  if (metrics.x_scale == 0) {
108  metrics.x_scale = metrics.y_scale;
109  } else {
110  if (metrics.y_scale == 0) {
111  metrics.y_scale = metrics.x_scale;
112  }
113  }
114 // goto Calculate_Ppem;
115  doall = false;
116 Debug(0, DebugTag.DBG_INIT, TAG, "met5: "+w+"!"+h+"!");
117  break;
118  case MAX:
119  break;
120  }
121  if (doall) {
122  /* to be on the safe side */
123  if (w < 0) {
124  w = -w;
125  }
126  if (h < 0) {
127  h = -h;
128  }
129  scaled_w = FTSizeRec.FT_REQUEST_WIDTH (req);
130  scaled_h = FTSizeRec.FT_REQUEST_HEIGHT(req);
131 Debug(0, DebugTag.DBG_INIT, TAG, "scaled_w: "+scaled_w+"!"+scaled_h+"!"+req);
132  /* determine scales */
133  if (req.width != 0) {
134  metrics.x_scale = FTCalc.FTDivFix(scaled_w, w);
135  if (req.height != 0) {
136  metrics.y_scale = FTCalc.FTDivFix(scaled_h, h);
137  if (req.type == FTTags.SizeRequestType.CELL) {
138  if (metrics.y_scale > metrics.x_scale) {
139  metrics.y_scale = metrics.x_scale;
140  } else {
141  metrics.x_scale = metrics.y_scale;
142  }
143  }
144  } else {
145  metrics.y_scale = metrics.x_scale;
146  scaled_h = FTCalc.FT_MulDiv(scaled_w, h, w);
147  }
148  } else {
149  metrics.x_scale = metrics.y_scale = FTCalc.FTDivFix(scaled_h, h);
150  scaled_w = FTCalc.FT_MulDiv(scaled_h, w, h);
151  }
152  }
153  /* calculate the ppems */
154  if (req.type != FTTags.SizeRequestType.NOMINAL) {
155  scaled_w = FTCalc.FTMulFix(face.units_per_EM, metrics.x_scale);
156  scaled_h = FTCalc.FTMulFix(face.units_per_EM, metrics.y_scale);
157  }
158  metrics.x_ppem = (short)((scaled_w + 32) >> 6);
159  metrics.y_ppem = (short)((scaled_h + 32) >> 6);
160 Debug(0, DebugTag.DBG_INIT, TAG, String.format("scaled: %d %d %d %d", scaled_w, scaled_h, metrics.x_ppem, metrics.y_ppem));
161  FTReference<FTSizeMetricsRec> metrics_ref = new FTReference<FTSizeMetricsRec>();
162  metrics_ref.Set(metrics);
163  ft_recompute_scaled_metrics(face, metrics_ref);
164  metrics = metrics_ref.Get();
165  } else {
166 // FT_ZERO( metrics );
167  metrics.x_scale = 1 << 16;
168  metrics.y_scale = 1 << 16;
169  }
170 int traceLevel = 7;
171  FTTrace.Trace(traceLevel, TAG, "FT_Request_Metrics:");
172  FTTrace.Trace(traceLevel, TAG, String.format(" x scale: %d (%f)",
173  metrics.x_scale, metrics.x_scale / 65536.0 ));
174  FTTrace.Trace(traceLevel, TAG, String.format(" y scale: %d (%f)",
175  metrics.y_scale, metrics.y_scale / 65536.0 ));
176  FTTrace.Trace(traceLevel, TAG, String.format(" ascender: %f", metrics.ascender / 64.0));
177  FTTrace.Trace(traceLevel, TAG, String.format(" descender: %f", metrics.descender / 64.0));
178  FTTrace.Trace(traceLevel, TAG, String.format(" height: %f", metrics.height / 64.0));
179  FTTrace.Trace(traceLevel, TAG, String.format(" max advance: %f", metrics.max_advance / 64.0));
180  FTTrace.Trace(traceLevel, TAG, String.format(" x ppem: %d", metrics.x_ppem ));
181  FTTrace.Trace(traceLevel, TAG, String.format(" y ppem: %d", metrics.y_ppem ));
182  return error;
183  }
