170  /* =====================================================================
171  * ft_raster1_render
172  * =====================================================================
173  */
174  private FTError.ErrorTag ft_raster1_render(FTRendererRec render, FTGlyphSlotRec slot, int mode, FTVectorRec origin) {
175  FTError.ErrorTag error = FTError.ErrorTag.ERR_OK;
176  FTOutlineRec outline;
177  FTBBoxRec cbox = new FTBBoxRec();
178  Integer width;
179  Integer height;
180  Integer pitch;
181  FTBitmapRec bitmap;
182  FTRasterParams params = new FTRasterParams();
183  FTReference<FTBBoxRec> cbox_ref = new FTReference<FTBBoxRec>();
184 
185  /* check glyph image format */
186  if (slot.format.getVal() != render.glyph_format.getVal()) {
187  error = FTError.ErrorTag.GLYPH_INVALID_ARGUMENT;
188  return error;
189  }
190  /* check rendering mode */
191  if (mode != FTTags.RenderMode.MONO.getVal()) {
192  /* raster1 is only capable of producing monochrome bitmaps */
193  if (render.module_clazz.module_name == "raster1") {
194  return FTError.ErrorTag.GLYPH_CANNOT_RENDER_GLYPH;
195  }
196  } else {
197  /* raster5 is only capable of producing 5-gray-levels bitmaps */
198  if (render.module_clazz.module_name == "raster5")
199  return FTError.ErrorTag.GLYPH_CANNOT_RENDER_GLYPH;
200  }
201  outline = slot.outline;
202  /* translate the outline to the new origin if needed */
203  FTReference<FTOutlineRec> outline_ref = new FTReference<FTOutlineRec>();
204  if (origin != null) {
205  outline_ref.Set(outline);
206  FTOutlineRec.FTOutlineTranslate(outline_ref, origin.x, origin.y);
207  outline = outline_ref.Get();
208  }
209  /* compute the control box, and grid fit it */
210  outline_ref.Set(outline);
211  cbox_ref.Set(cbox);
212  FTOutlineRec.FTOutlineGetCBox(outline_ref, cbox_ref);
213  cbox = cbox_ref.Get();
214  outline = outline_ref.Get();
215  /* undocumented but confirmed: bbox values get rounded */
216  cbox.xMin = FTCalc.FT_PIX_ROUND(cbox.xMin);
217  cbox.yMin = FTCalc.FT_PIX_ROUND(cbox.yMin);
218  cbox.xMax = FTCalc.FT_PIX_ROUND(cbox.xMax);
219  cbox.yMax = FTCalc.FT_PIX_ROUND(cbox.yMax);
220  width = (int)((cbox.xMax - cbox.xMin) >> 6);
221  height = (int)((cbox.yMax - cbox.yMin) >> 6);
222  if (width > 0xFFFF || height > 0xFFFF) {
223  error = FTError.ErrorTag.GLYPH_INVALID_ARGUMENT;
224  return error;
225  }
226  bitmap = slot.bitmap;
227  /* release old bitmap buffer */
228  if ((slot.internal.flags.getVal() & FTTags.GlyphFormat.OWN_BITMAP.getVal()) != 0) {
229 // FT_FREE(bitmap.buffer);
230  slot.internal.flags = FTTags.GlyphFormat.getTableTag(slot.internal.flags.getVal() & ~FTTags.GlyphFormat.OWN_BITMAP.getVal());
231  }
232  /* allocate new one, depends on pixel format */
233  if ((mode & FTTags.RenderMode.MONO.getVal()) == 0) {
234  /* we pad to 32 bits, only for backwards compatibility with FT 1.x */
235  pitch = FTCalc.FT_PAD_CEIL(width, 4);
236  bitmap.pixel_mode = FTTags.PixelMode.GRAY;
237  bitmap.num_grays = 256;
238  } else {
239  pitch = ((width + 15) >> 4) << 1;
240  bitmap.pixel_mode = FTTags.PixelMode.MONO;
241  }
242  bitmap.width = width;
243  bitmap.rows = height;
244  bitmap.pitch = pitch;
245  // realloc cout = pitch, item_size ) height
246  bitmap.buffer = new byte[pitch * height];
247  if (bitmap.buffer == null) {
248  return error;
249  }
250  slot.internal.flags = FTTags.GlyphFormat.getTableTag(slot.internal.flags.getVal() | FTTags.GlyphFormat.OWN_BITMAP.getVal());
251  Debug(0, DebugTag.DBG_INIT, TAG, String.format("cbox_xMin: %d %dn", -cbox.xMin, -cbox.yMin));
252  /* translate outline to render it into the bitmap */
253  outline_ref.Set(outline);
254  FTOutlineRec.FTOutlineTranslate(outline_ref, -cbox.xMin, -cbox.yMin);
255  outline = outline_ref.Get();
256  /* set up parameters */
257  params.target = bitmap;
258  params.source = outline;
259  params.flags = 0;
260  if (bitmap.pixel_mode == FTTags.PixelMode.GRAY) {
261  params.flags |= FTRasterParams.FT_RASTER_FLAG_AA;
262  }
263  /* render outline into the bitmap */
264  error = render.rasterRender(render.raster, params);
265  outline_ref.Set(outline);
266  FTOutlineRec.FTOutlineTranslate(outline_ref, cbox.xMin, cbox.yMin);
267  outline = outline_ref.Get();
268  if (error != FTError.ErrorTag.ERR_OK) {
269  return error;
270  }
271  slot.format = FTTags.GlyphFormat.BITMAP;
272  slot.bitmap_left = (int)(cbox.xMin >> 6);
273  slot.bitmap_top = (int)(cbox.yMax >> 6);
274  return error;
275  }
