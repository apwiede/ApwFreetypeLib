83  /* ==================== Load ===================================== */
84  public FTError.ErrorTag Load(FTStreamRec stream) {
85  long offset = stream.pos();
86  int tag = stream.readInt();
87  this.tag = TTTags.Table.getTableTag((int)tag);
88 Debug(0, DebugTag.DBG_INIT, TAG, "Tag: " + String.format("0x%08x offset: 0x%08x", tag, offset));
89  if (tag != 0x00010000 &&
90  tag != TTTags.Table.ttcf.getVal() &&
91  tag != TTTags.Table.OTTO.getVal() &&
92  tag != TTTags.Table.truE.getVal() &&
93  tag != TTTags.Table.typ1.getVal() &&
94  tag != 0x00020000) {
95  FTTrace.Trace(7, TAG, " not a font using the SFNT container format");
96  return FTError.ErrorTag.LOAD_UNKNOWN_FILE_FORMAT;
97  }
98  this.tag = TTTags.Table.ttcf;
99  if (tag == TTTags.Table.ttcf.getVal()) {
100  int n;
101 
102  FTTrace.Trace(7, TAG, "sfnt_open_font: file is a collection");
103  this.version = stream.readInt();
104  this.count = stream.readInt();
105 
106  if (this.count == 0) {
107  return FTError.ErrorTag.LOAD_INVALID_TABLE;
108  }
109  /* a rough size estimate: let's conservatively assume that there */
110  /* is just a single table info in each subfont header (12 + 16*1 = */
111  /* 28 bytes), thus we have (at least) `12 + 4*count' bytes for the */
112  /* size of the TTC header plus `28*count' bytes for all subfont */
113  /* headers */
114  if (this.count > stream.getSize() / (28 + 4)) {
115  return FTError.ErrorTag.LOAD_ARRAY_TOO_LARGE;
116  }
117  /* now read the offsets of each font in the file */
118  this.offsets = new long[this.count];
119  for (n = 0; n < this.count; n++) {
120  this.offsets[n] = stream.readInt();
121  }
122  } else {
123  FTTrace.Trace(7, TAG, "sfnt_open_font: synthesize TTC");
124  this.version = (1 << 16);
125  this.count = 1;
126  this.offsets = new long[1];
127  this.offsets[0] = offset;
128  }
129  return FTError.ErrorTag.ERR_OK;
130  }
