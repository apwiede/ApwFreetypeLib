120  public static int TTMulFix14(int a, int b) {
121  int sign;
122  int ah;
123  int al;
124  int mid;
125  int lo;
126  int hi;
127 
128  sign = (int)(a ^ b);
129  if (a < 0) {
130  a = -a;
131  }
132  if (b < 0) {
133  b = -b;
134  }
135  ah = ((a >> 16) & 0xFFFF);
136  al = (a & 0xFFFF);
137  lo = al * b;
138  mid = ah * b;
139  hi = mid >> 16;
140  mid = (mid << 16) + (1 << 13); /* rounding */
141  lo += mid;
142  if (lo < mid) {
143  hi += 1;
144  }
145  mid = ((lo >> 14) | (hi << 18));
146  return sign >= 0 ? mid : -mid;
147  }
