186  /* =====================================================================
187  * FT_LOAD_XXX
188  *
189  * @description:
190  * A list of bit-field constants used with @FT_Load_Glyph to indicate
191  * what kind of operations to perform during glyph loading.
192  *
193  * @values:
194  * FT_LOAD_DEFAULT ::
195  * Corresponding to~0, this value is used as the default glyph load
196  * operation. In this case, the following happens:
197  *
198  * 1. FreeType looks for a bitmap for the glyph corresponding to the
199  * face's current size. If one is found, the function returns.
200  * The bitmap data can be accessed from the glyph slot (see note
201  * below).
202  *
203  * 2. If no embedded bitmap is searched or found, FreeType looks for a
204  * scalable outline. If one is found, it is loaded from the font
205  * file, scaled to device pixels, then `hinted' to the pixel grid
206  * in order to optimize it. The outline data can be accessed from
207  * the glyph slot (see note below).
208  *
209  * Note that by default, the glyph loader doesn't render outlines into
210  * bitmaps. The following flags are used to modify this default
211  * behaviour to more specific and useful cases.
212  *
213  * FT_LOAD_NO_SCALE ::
214  * Don't scale the loaded outline glyph but keep it in font units.
215  *
216  * This flag implies @FT_LOAD_NO_HINTING and @FT_LOAD_NO_BITMAP, and
217  * unsets @FT_LOAD_RENDER.
218  *
219  * If the font is `tricky' (see @FT_FACE_FLAG_TRICKY for more), using
220  * FT_LOAD_NO_SCALE usually yields meaningless outlines because the
221  * subglyphs must be scaled and positioned with hinting instructions.
222  * This can be solved by loading the font without FT_LOAD_NO_SCALE and
223  * setting the character size to `font->units_per_EM'.
224  *
225  * FT_LOAD_NO_HINTING ::
226  * Disable hinting. This generally generates `blurrier' bitmap glyphs
227  * when the glyph are rendered in any of the anti-aliased modes. See
228  * also the note below.
229  *
230  * This flag is implied by @FT_LOAD_NO_SCALE.
231  *
232  * FT_LOAD_RENDER ::
233  * Call @FT_Render_Glyph after the glyph is loaded. By default, the
234  * glyph is rendered in @FT_RENDER_MODE_NORMAL mode. This can be
235  * overridden by @FT_LOAD_TARGET_XXX or @FT_LOAD_MONOCHROME.
236  *
237  * This flag is unset by @FT_LOAD_NO_SCALE.
238  *
239  * FT_LOAD_NO_BITMAP ::
240  * Ignore bitmap strikes when loading. Bitmap-only fonts ignore this
241  * flag.
242  *
243  * @FT_LOAD_NO_SCALE always sets this flag.
244  *
245  * FT_LOAD_VERTICAL_LAYOUT ::
246  * Load the glyph for vertical text layout. In particular, the
247  * `advance' value in the @FT_GlyphSlotRec structure is set to the
248  * `vertAdvance' value of the `metrics' field.
249  *
250  * In case @FT_HAS_VERTICAL doesn't return true, you shouldn't use
251  * this flag currently. Reason is that in this case vertical metrics
252  * get synthesized, and those values are not always consistent across
253  * various font formats.
254  *
255  * FT_LOAD_FORCE_AUTOHINT ::
256  * Indicates that the auto-hinter is preferred over the font's native
257  * hinter. See also the note below.
258  *
259  * FT_LOAD_CROP_BITMAP ::
260  * Indicates that the font driver should crop the loaded bitmap glyph
261  * (i.e., remove all space around its black bits). Not all drivers
262  * implement this.
263  *
264  * FT_LOAD_PEDANTIC ::
265  * Indicates that the font driver should perform pedantic verifications
266  * during glyph loading. This is mostly used to detect broken glyphs
267  * in fonts. By default, FreeType tries to handle broken fonts also.
268  *
269  * In particular, errors from the TrueType bytecode engine are not
270  * passed to the application if this flag is not set; this might
271  * result in partially hinted or distorted glyphs in case a glyph's
272  * bytecode is buggy.
273  *
274  * FT_LOAD_IGNORE_GLOBAL_ADVANCE_WIDTH ::
275  * Ignored. Deprecated.
276  *
277  * FT_LOAD_NO_RECURSE ::
278  * This flag is only used internally. It merely indicates that the
279  * font driver should not load composite glyphs recursively. Instead,
280  * it should set the `num_subglyph' and `subglyphs' values of the
281  * glyph slot accordingly, and set `glyph->format' to
282  * @FT_GLYPH_FORMAT_COMPOSITE.
283  *
284  * The description of sub-glyphs is not available to client
285  * applications for now.
286  *
287  * This flag implies @FT_LOAD_NO_SCALE and @FT_LOAD_IGNORE_TRANSFORM.
288  *
289  * FT_LOAD_IGNORE_TRANSFORM ::
290  * Indicates that the transform matrix set by @FT_Set_Transform should
291  * be ignored.
292  *
293  * FT_LOAD_MONOCHROME ::
294  * This flag is used with @FT_LOAD_RENDER to indicate that you want to
295  * render an outline glyph to a 1-bit monochrome bitmap glyph, with
296  * 8~pixels packed into each byte of the bitmap data.
297  *
298  * Note that this has no effect on the hinting algorithm used. You
299  * should rather use @FT_LOAD_TARGET_MONO so that the
300  * monochrome-optimized hinting algorithm is used.
301  *
302  * FT_LOAD_LINEAR_DESIGN ::
303  * Indicates that the `linearHoriAdvance' and `linearVertAdvance'
304  * fields of @FT_GlyphSlotRec should be kept in font units. See
305  * @FT_GlyphSlotRec for details.
306  *
307  * FT_LOAD_NO_AUTOHINT ::
308  * Disable auto-hinter. See also the note below.
309  *
310  * FT_LOAD_COLOR ::
311  * This flag is used to request loading of color embedded-bitmap
312  * images. The resulting color bitmaps, if available, will have the
313  * @FT_PIXEL_MODE_BGRA format. When the flag is not used and color
314  * bitmaps are found, they will be converted to 256-level gray
315  * bitmaps transparently. Those bitmaps will be in the
316  * @FT_PIXEL_MODE_GRAY format.
317  *
318  * @note:
319  * By default, hinting is enabled and the font's native hinter (see
320  * @FT_FACE_FLAG_HINTER) is preferred over the auto-hinter. You can
321  * disable hinting by setting @FT_LOAD_NO_HINTING or change the
322  * precedence by setting @FT_LOAD_FORCE_AUTOHINT. You can also set
323  * @FT_LOAD_NO_AUTOHINT in case you don't want the auto-hinter to be
324  * used at all.
325  *
326  * See the description of @FT_FACE_FLAG_TRICKY for a special exception
327  * (affecting only a handful of Asian fonts).
328  *
329  * Besides deciding which hinter to use, you can also decide which
330  * hinting algorithm to use. See @FT_LOAD_TARGET_XXX for details.
331  *
332  * Note that the auto-hinter needs a valid Unicode cmap (either a native
333  * one or synthesized by FreeType) for producing correct results. If a
334  * font provides an incorrect mapping (for example, assigning the
335  * character code U+005A, LATIN CAPITAL LETTER Z, to a glyph depicting a
336  * mathematical integral sign), the auto-hinter might produce useless
337  * results.
338  *
339  * Bits 16..19 are used by `FT_LOAD_TARGET_'
340  * =====================================================================
341  */

424  /* ===================================================================== */
425  /* FT_OUTLINE_FLAGS */
426  /* */
427  /* <Description> */
428  /* A list of bit-field constants use for the flags in an outline's */
429  /* `flags' field. */
430  /* */
431  /* <Values> */
432  /* FT_OUTLINE_NONE :: */
433  /* Value~0 is reserved. */
434  /* */
435  /* FT_OUTLINE_OWNER :: */
436  /* If set, this flag indicates that the outline's field arrays */
437  /* (i.e., `points', `flags', and `contours') are `owned' by the */
438  /* outline object, and should thus be freed when it is destroyed. */
439  /* */
440  /* FT_OUTLINE_EVEN_ODD_FILL :: */
441  /* By default, outlines are filled using the non-zero winding rule. */
442  /* If set to 1, the outline will be filled using the even-odd fill */
443  /* rule (only works with the smooth rasterizer). */
444  /* */
445  /* FT_OUTLINE_REVERSE_FILL :: */
446  /* By default, outside contours of an outline are oriented in */
447  /* clock-wise direction, as defined in the TrueType specification. */
448  /* This flag is set if the outline uses the opposite direction */
449  /* (typically for Type~1 fonts). This flag is ignored by the scan */
450  /* converter. */
451  /* */
452  /* FT_OUTLINE_IGNORE_DROPOUTS :: */
453  /* By default, the scan converter will try to detect drop-outs in */
454  /* an outline and correct the glyph bitmap to ensure consistent */
455  /* shape continuity. If set, this flag hints the scan-line */
456  /* converter to ignore such cases. See below for more information. */
457  /* */
458  /* FT_OUTLINE_SMART_DROPOUTS :: */
459  /* Select smart dropout control. If unset, use simple dropout */
460  /* control. Ignored if @FT_OUTLINE_IGNORE_DROPOUTS is set. See */
461  /* below for more information. */
462  /* */
463  /* FT_OUTLINE_INCLUDE_STUBS :: */
464  /* If set, turn pixels on for `stubs', otherwise exclude them. */
465  /* Ignored if @FT_OUTLINE_IGNORE_DROPOUTS is set. See below for */
466  /* more information. */
467  /* */
468  /* FT_OUTLINE_HIGH_PRECISION :: */
469  /* This flag indicates that the scan-line converter should try to */
470  /* convert this outline to bitmaps with the highest possible */
471  /* quality. It is typically set for small character sizes. Note */
472  /* that this is only a hint that might be completely ignored by a */
473  /* given scan-converter. */
474  /* */
475  /* FT_OUTLINE_SINGLE_PASS :: */
476  /* This flag is set to force a given scan-converter to only use a */
477  /* single pass over the outline to render a bitmap glyph image. */
478  /* Normally, it is set for very large character sizes. It is only */
479  /* a hint that might be completely ignored by a given */
480  /* scan-converter. */
481  /* */
482  /* <Note> */
483  /* The flags @FT_OUTLINE_IGNORE_DROPOUTS, @FT_OUTLINE_SMART_DROPOUTS, */
484  /* and @FT_OUTLINE_INCLUDE_STUBS are ignored by the smooth */
485  /* rasterizer. */
486  /* */
487  /* There exists a second mechanism to pass the drop-out mode to the */
488  /* B/W rasterizer; see the `tags' field in @FT_Outline. */
489  /* */
490  /* Please refer to the description of the `SCANTYPE' instruction in */
491  /* the OpenType specification (in file `ttinst1.doc') how simple */
492  /* drop-outs, smart drop-outs, and stubs are defined. */
493  /* */
494  /* ===================================================================== */
