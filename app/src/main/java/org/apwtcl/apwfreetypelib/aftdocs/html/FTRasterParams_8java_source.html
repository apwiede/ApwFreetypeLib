18  /* */
19  /* <Description> */
20  /* A structure to hold the arguments used by a raster's render */
21  /* function. */
22  /* */
23  /* <Fields> */
24  /* target :: The target bitmap. */
25  /* */
26  /* source :: A pointer to the source glyph image (e.g., an */
27  /* @FT_Outline). */
28  /* */
29  /* flags :: The rendering flags. */
30  /* */
31  /* gray_spans :: The gray span drawing callback. */
32  /* */
33  /* black_spans :: The black span drawing callback. UNIMPLEMENTED! */
34  /* */
35  /* bit_test :: The bit test callback. UNIMPLEMENTED! */
36  /* */
37  /* bit_set :: The bit set callback. UNIMPLEMENTED! */
38  /* */
39  /* user :: User-supplied data that is passed to each drawing */
40  /* callback. */
41  /* */
42  /* clip_box :: An optional clipping box. It is only used in */
43  /* direct rendering mode. Note that coordinates here */
44  /* should be expressed in _integer_ pixels (and not in */
45  /* 26.6 fixed-point units). */
46  /* */
47  /* <Note> */
48  /* An anti-aliased glyph bitmap is drawn if the @FT_RASTER_FLAG_AA */
49  /* bit flag is set in the `flags' field, otherwise a monochrome */
50  /* bitmap is generated. */
51  /* */
52  /* If the @FT_RASTER_FLAG_DIRECT bit flag is set in `flags', the */
53  /* raster will call the `gray_spans' callback to draw gray pixel */
54  /* spans, in the case of an aa glyph bitmap, it will call */
55  /* `black_spans', and `bit_test' and `bit_set' in the case of a */
56  /* monochrome bitmap. This allows direct composition over a */
57  /* pre-existing bitmap through user-provided callbacks to perform the */
58  /* span drawing/composition. */
59  /* */
60  /* Note that the `bit_test' and `bit_set' callbacks are required when */
61  /* rendering a monochrome bitmap, as they are crucial to implement */
62  /* correct drop-out control as defined in the TrueType specification. */
63  /* */

67  /* FT_RASTER_FLAG_XXX */
68  /* */
69  /* <Description> */
70  /* A list of bit flag constants as used in the `flags' field of a */
71  /* @FT_Raster_Params structure. */
72  /* */
73  /* <Values> */
74  /* FT_RASTER_FLAG_DEFAULT :: This value is 0. */
75  /* */
76  /* FT_RASTER_FLAG_AA :: This flag is set to indicate that an */
77  /* anti-aliased glyph image should be */
78  /* generated. Otherwise, it will be */
79  /* monochrome (1-bit). */
80  /* */
81  /* FT_RASTER_FLAG_DIRECT :: This flag is set to indicate direct */
82  /* rendering. In this mode, client */
83  /* applications must provide their own span */
84  /* callback. This lets them directly */
85  /* draw or compose over an existing bitmap. */
86  /* If this bit is not set, the target */
87  /* pixmap's buffer _must_ be zeroed before */
88  /* rendering. */
89  /* */
90  /* Note that for now, direct rendering is */
91  /* only possible with anti-aliased glyphs. */
92  /* */
93  /* FT_RASTER_FLAG_CLIP :: This flag is only used in direct */
94  /* rendering mode. If set, the output will */
95  /* be clipped to a box specified in the */
96  /* `clip_box' field of the */
97  /* @FT_Raster_Params structure. */
98  /* */
99  /* Note that by default, the glyph bitmap */
100  /* is clipped to the target pixmap, except */
101  /* in direct rendering mode where all spans */
102  /* are generated if no clipping box is set. */
103  /* */
